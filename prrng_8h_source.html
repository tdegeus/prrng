402 {

895 uint32_t next_uint32(uint32_t bound)
896 {
897 // To avoid bias, we need to make the range of the RNG a multiple of
898 // bound, which we do by dropping output less than a threshold.
899 // A naive scheme to calculate the threshold would be to do
900 //
901 // uint32_t threshold = 0x100000000ull % bound;
902 //
903 // but 64-bit div/mod is slower than 32-bit div/mod (especially on
904 // 32-bit platforms). In essence, we do
905 //
906 // uint32_t threshold = (0x100000000ull-bound) % bound;
907 //
908 // because this version will calculate the same modulus, but the LHS
909 // value is less than 2^32.
910
911 uint32_t threshold = (~bound + 1u) % bound;
912
913 // Uniformity guarantees that this loop will terminate. In practice, it
914 // should usually terminate quickly; on average (assuming all bounds are
915 // equally likely), 82.25% of the time, we can expect it to require just
916 // one iteration. In the worst case, someone passes a bound of 2^31 + 1
917 // (i.e., 2147483649), which invalidates almost 50% of the range. In
918 // practice, bounds are typically small and only a tiny amount of the range
919 // is eliminated.
920 for (;;) {
921 uint32_t r = next_uint32();
922 if (r >= threshold) {
923 return r % bound;
924 }
925 }
